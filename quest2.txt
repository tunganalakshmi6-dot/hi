    <script>
        let user = JSON.parse(localStorage.getItem('diet_user')) || null;
        let macros = JSON.parse(localStorage.getItem('diet_macros')) || { cal: 0, prot: 0 };
        let timerInterval;
        let timeLeft = 300; 

        window.onload = function() {
            if (user) { showDashboard(); }
        }

        function handleLogin(e) {
            e.preventDefault();
            user = {
                name: document.getElementById('username').value,
                targetCal: parseInt(document.getElementById('target-cal').value),
                targetProt: parseInt(document.getElementById('target-prot').value),
                streak: 1
            };
            localStorage.setItem('diet_user', JSON.stringify(user));
            showDashboard();
        }

        function showDashboard() {
            document.getElementById('auth-section').classList.add('hidden');
            document.getElementById('dashboard').classList.remove('hidden');
            document.getElementById('nav-user').classList.remove('hidden');
            
            document.getElementById('user-display-name').innerText = user.name;
            document.getElementById('streak-badge').innerText = `🔥 Streak: ${user.streak} days`;
            document.getElementById('streak-count').innerText = `${user.streak} Day${user.streak > 1 ? 's' : ''}`;
            updateMacroUI();
        }

        function logout() {
            localStorage.clear();
            user = null;
            macros = { cal: 0, prot: 0 };
            document.getElementById('auth-section').classList.remove('hidden');
            document.getElementById('dashboard').classList.add('hidden');
            document.getElementById('nav-user').classList.add('hidden');
            resetTimer();
        }

        function addMacro(type) {
            const valInput = document.getElementById(`add-${type}-val`);
            const value = parseInt(valInput.value) || 0;
            if (value > 0) {
                macros[type] += value;
                localStorage.setItem('diet_macros', JSON.stringify(macros));
                updateMacroUI();
                valInput.value = '';
            }
        }

        function updateMacroUI() {
            document.getElementById('cal-ratio').innerText = `${macros.cal} / ${user.targetCal} kcal`;
            document.getElementById('prot-ratio').innerText = `${macros.prot} / ${user.targetProt} g`;
            
            const calPct = Math.min((macros.cal / user.targetCal) * 100, 100);
            const protPct = Math.min((macros.prot / user.targetProt) * 100, 100);
            
            document.getElementById('cal-progress').style.width = `${calPct}%`;
            document.getElementById('prot-progress').style.width = `${protPct}%`;
        }

        function toggleTask() {}

        function calculateBMI() {
            const w = parseFloat(document.getElementById('bmi-weight').value);
            const h = parseFloat(document.getElementById('bmi-height').value) / 100; 
            const box = document.getElementById('bmi-result');

            if (w > 0 && h > 0) {
                const bmi = (w / (h * h)).toFixed(1);
                box.classList.remove('hidden', 'bg-emerald-100', 'text-emerald-800', 'bg-amber-100', 'text-amber-800', 'bg-rose-100', 'text-rose-800');
                
                let category = "Normal";
                let style = "bg-emerald-100 text-emerald-800";
                
                if (bmi < 18.5) { category = "Underweight"; style = "bg-amber-100 text-amber-800"; }
                else if (bmi >= 25 && bmi < 30) { category = "Overweight"; style = "bg-amber-100 text-amber-800"; }
                else if (bmi >= 30) { category = "Obese"; style = "bg-rose-100 text-rose-800"; }
                
                box.className = `text-center p-2.5 rounded-lg text-sm font-semibold ${style}`;
                box.innerText = `Your BMI is ${bmi} (${category})`;
            }
        }

        function updateTimerUI() {
            const mins = Math.floor(timeLeft / 60).toString().padStart(2, '0');
            const secs = (timeLeft % 60).toString().padStart(2, '0');
            document.getElementById('timer-display').innerText = `${mins}:${secs}`;
        }

        function startTimer() {
            if (!timerInterval) {
                timerInterval = setInterval(() => {
                    if (timeLeft > 0) {
                        timeLeft--;
                        updateTimerUI();
                    } else {
                        clearInterval(timerInterval);
                        alert("Time is up!");
                    }
                }, 1000);
            }
        }

        function pauseTimer() {
            clearInterval(timerInterval);
            timerInterval = null;
        }

        function resetTimer() {
            pauseTimer();
            timeLeft = 300;
            updateTimerUI();
        }

        function toggleModal(id) {
            const el = document.getElementById(id);
            el.classList.toggle('hidden');
        }
    </script>
